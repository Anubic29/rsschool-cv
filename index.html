public static Elem[] Search(Elem[] mas)
{
    int k = mas.Length;
    Elem[] lmas = new Elem[2 * k];
    lmas[k - 1] = mas[0];
    int pos = k - 1;
    for (int i = 1; i < k; i++)
    {
        int j = 0;
        if (mas[i].key >= lmas[k - 1].key)
        {
            j++;
            while (lmas[k - 1 + j] != null && mas[i].key > lmas[k - 1 + j].key)
                j++;
        }
        else
        {
            j--;
            pos--;

                j--;
        }
